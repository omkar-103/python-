Check Wheteher Prime or Not

def test_prime():
    n = int(input("Enter the number: "))
    if n == 1:
        return False
    elif n == 2:
        return True
    else:
        for i in range(2, n):
            if n % i == 0:
                return False
        return True

print(test_prime())

to Find the power of a number using recursion

def power(base, exponent):
    if exponent == 0:
        return 1
    else:
        return base * power(base, exponent - 1)

base = float(input("Enter the base: "))
exponent = int(input("Enter the exponent: "))
result = power(base, exponent)
print(f"{base} raised to the power of {exponent} is {result}")

Using a lambda function with reduce function to compute maximum element from list.

from functools import reduce

numbers = [10, 5, 20, 15, 30]

max_number = reduce(lambda x, y: x if x > y else y, numbers)

print(f"The maximum number is: {max_number}")

Find factorial using recursion.

def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n - 1)

num = int(input("Enter a number: "))
result = factorial(num)
print(f"The factorial of {num} is {result}")

Display the Fibonacci sequence. 

def fibonacci(n):
    fib_sequence = [0, 1]
    for i in range(2, n):
        next_fib = fib_sequence[-1] + fib_sequence[-2]
        fib_sequence.append(next_fib)
    return fib_sequence

num_terms = int(input("Enter the number of terms: "))
fibonacci_sequence = fibonacci(num_terms)
print("Fibonacci Sequence:")
print(fibonacci_sequence)

Get execution time of a python method in seconds. 

import time

def sum_of_n_number(n):
    start_time = time.time()
    s = 0
    for i in range(1, n + 1):
        s = s + i
    end_time = time.time()
    return s, end_time - start_time

n = 10
result, execution_time = sum_of_n_number(n)
print(f"Sum of numbers from 1 to {n} is {result}")
print(f"Time taken to calculate: {execution_time:.6f} seconds")

Count the occurrences of each word in the given sentences.

def wordcount(str):
    counts = dict()
    words = str.split()
    for i in words:
        if i in counts:
            counts[i] += 1
        else:
            counts[i] = 1
    return counts

print(wordcount('growth through excellence'))

To reverse a string if its length is multiple of 4.

def reverse(str1):
    if len(str1) % 4 == 0:
        return ''.join(reversed(str1))
    return str1

print(reverse('notification'))
print(reverse('intrduction'))
